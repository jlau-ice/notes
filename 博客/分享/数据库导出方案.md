
```bash
docker run --rm -i mysql:8.0 \
mysqldump \
-h [线上服务器公网IP] \
-P [线上MySQL端口，默认3306] \
-u [线上用户名] \
-p'线上密码' \
--databases cloud_cart cloud_item cloud_pay cloud_trade cloud_user \
--single-transaction \
--routines \
--triggers \
--events > online_data_backup.sql
```

-p'xxxxxx' \
