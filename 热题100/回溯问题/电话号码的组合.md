[题目地址](https://leetcode.cn/problems/letter-combinations-of-a-phone-number)

常规的回溯解法，要注意 index ,这里index 指的是到第几个数字了，指的不是数字里面的字母。
```java
class Solution {
    List<String> res = new ArrayList<>();
    StringBuilder sb = new StringBuilder();

    public List<String> letterCombinations(String digits) {
        Map<Character, String> map = new HashMap<>();
        map.put('2', "abc");
        map.put('3', "def");
        map.put('4', "ghi");
        map.put('5', "jkl");
        map.put('6', "mno");
        map.put('7', "pqrs");
        map.put('8', "tuv");
        map.put('9', "wxyz");
        backtracking(digits, map, 0);
        return res;
    }

    private void backtracking(String digits, Map<Character, String> map, int index) {
        if (sb.length() == digits.length()) {
            res.add(new String(sb));
            return;
        }
        char c = digits.charAt(index);
        String s = map.get(c);
        for (int i = 0; i < s.length(); i++) {
            sb.append(s.charAt(i));
            backtracking(digits, map, index + 1);
            sb.deleteCharAt(sb.length() - 1);
        }
    }
}
```
