[题目地址](https://leetcode.cn/problems/convert-sorted-array-to-binary-search-tree)
有点像二分，从大数组中见元素作为根节点。剩下两部分的中间 作为左右孩子的根节点，以此类推，递归调用。
```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public TreeNode sortedArrayToBST(int[] nums) {
        return mid(nums, 0, nums.length - 1);
    }

    public TreeNode mid(int[] nums, int start, int end) {
        if (start > end) {
            return null;
        }

        int m = (end - start) / 2 + start;
        TreeNode root = new TreeNode(nums[m]);
        root.left = mid(nums, start, m - 1);
        root.right = mid(nums, m + 1, end);
        return root;
    }

}
```
