找到问题了，必须严格递增。最小值设置比int 最小值还小，防止老6用int最小值当节点。
```java
class Solution {
    public boolean isValidBST(TreeNode root) {
        Deque<TreeNode> stack = new ArrayDeque<>();
        TreeNode cur = root;
        long lastNodeVal = Long.MIN_VALUE;
        while (!stack.isEmpty() || cur != null) {
            if (cur != null) {
                stack.push(cur);
                cur = cur.left;
            } else {
                cur = stack.pop();
                if (cur.val > lastNodeVal) {
                    lastNodeVal = cur.val;
                } else {
                    return false;
                }
                cur = cur.right;
            }
        }
        return true;
    }
}
```
