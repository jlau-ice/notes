class Solution {
    public void rotate(int[] nums, int k) {
        reverse(nums, 0, nums.length - 1);
        reverse(nums, 0, (k % nums.length) - 1);
        reverse(nums, (k % nums.length), nums.length - 1);
    }

    // [] 区间
    public void reverse(int[] arr, int start, int end) {
        while (start < end) {
            int temp = arr[start];
            arr[start] = arr[end];
            arr[end] = temp;
            start++;Q
            end--;
        }
    }
}
